# Integer
a = 10
print(a)

# Float
b = 10.5
print(b)

# String
c = "Hello, World!"
print(c)

# Boolean
d = True
print(d)

# List
e = [1, 2, 3, 4, 5]
print(e)

# Tuple
f = (1, 2, 3, 4, 5)
print(f)

# Dictionary
g = {"name": "Alice", "age": 25}
print(g)

# Set
h = {1, 2, 3, 4, 5}
print(h)

# NoneType
none_value = None
print(f"NoneType: {none_value}")
